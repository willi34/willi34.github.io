# Simple class to manage a book
class Book
  attr_accessor :title, :author

  def initialize(title, author)
    @title = title
    @author = author
  end

  def description
    "Book: #{@title} by #{@author}"
  end
end

# Example usage
book = Book.new("Ruby Basics", "Jane Doe")
puts book.description
